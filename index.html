k8nvas — helm install
# Add k8nvas chart repo
helm repo add k8nvas https://charts.k8nvas.io
 
helm install k8nvas k8nvas/agent \
  --set clusterName="production-us" \
  --set apiKey=$K8NVAS_KEY \
  -n kube-system
